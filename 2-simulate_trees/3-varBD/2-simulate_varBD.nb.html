  tree <- withTimeout(sim.bdsky.stt(n           = 0,
                                   timestop    = age,

                                   deathsky    = rev(RTT$death_rate),
                                   timesky     = seq(0.0, age, length.out=200),
                                   sampprobsky = rep(0,200),
